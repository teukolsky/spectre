24 template <typename T>

26  template <typename UniformRandomBitGenerator,

28  static void apply(
29  const gsl::not_null<T*> data,
30  const gsl::not_null<UniformRandomBitGenerator*> generator,
31  const gsl::not_null<RandomNumberDistribution*> distribution) noexcept {
32  static_assert(

85 /// \ingroup TestingFrameworkGroup
86 /// \brief Make a data structure and fill it with random values
87 ///
88 /// \details Given an object of type `T`, create an object of type `ReturnType`
89 /// whose elements are initialized to random values using the given random
90 /// number generator and random number distribution.
91 ///
92 /// \requires the type `ReturnType` to be creatable using
93 /// `make_with_value<ReturnType>(T)`
94 template <typename ReturnType, typename T, typename UniformRandomBitGenerator,
95  typename RandomNumberDistribution>
96 ReturnType make_with_random_values(
97  const gsl::not_null<UniformRandomBitGenerator*> generator,
98  const gsl::not_null<RandomNumberDistribution*> distribution,
99  const T& used_for_size) noexcept {
100  auto result = make_with_value<ReturnType>(
101  used_for_size, std::numeric_limits<double>::signaling_NaN());
102  fill_with_random_values(make_not_null(&result), generator, distribution);
103  return result;
104 }
