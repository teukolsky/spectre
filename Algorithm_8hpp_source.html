147 template <typename Invokable, typename InitialDataBox, typename ThisVariant,
148  typename... Variants, typename... Args,
149  Requires<is_apply_callable_v<
150  Invokable, std::add_lvalue_reference_t<ThisVariant>, Args&&...>> =
151  nullptr>
152 void apply_visitor_helper(boost::variant<Variants...>& box,
153  const gsl::not_null<int*> iter,
154  const gsl::not_null<bool*> already_visited,
155  Args&&... args) {
156  if (box.which() == *iter and not*already_visited) {
157  try {
158  make_overloader(
159  [&box](std::true_type /*returns_void*/, auto&&... my_args) {
160  Invokable::apply(boost::get<ThisVariant>(box),
161  std::forward<Args>(my_args)...);
162  },
163  [&box](std::false_type /*returns_void*/, auto&&... my_args) {
164  box = std::get<0>(Invokable::apply(boost::get<ThisVariant>(box),
165  std::forward<Args>(my_args)...));
166  using return_box_type = decltype(std::get<0>(Invokable::apply(
167  boost::get<ThisVariant>(box), std::forward<Args>(my_args)...)));
168  static_assert(
169  cpp17::is_same_v<std::decay_t<return_box_type>,
170  InitialDataBox> and
171  cpp17::is_same_v<db::DataBox<tmpl::list<>>, ThisVariant>,
172  "A simple action must return either void or take an empty "
173  "DataBox and return the initial_databox set in the parallel "
174  "component.");
175  })(
176  typename std::is_same<void, decltype(Invokable::apply(
177  std::declval<ThisVariant&>(),
178  std::declval<Args>()...))>::type{},
179  std::forward<Args>(args)...);
180 
181  } catch (std::exception& e) {
182  ERROR("Fatal error: Failed to call single Action '"
183  << pretty_type::get_name<Invokable>() << "' on iteration '" << iter
184  << "' with DataBox type '" << pretty_type::get_name<ThisVariant>()
185  << "'\nThe exception is: '" << e.what() << "'\n");
186  }
187  *already_visited = true;
188  }
189  (*iter)++;
190 }

192 template <typename Invokable, typename InitialDataBox, typename ThisVariant,
193  typename... Variants, typename... Args,
194  Requires<not is_apply_callable_v<
195  Invokable, std::add_lvalue_reference_t<ThisVariant>, Args&&...>> =
196  nullptr>
197 void apply_visitor_helper(boost::variant<Variants...>& box,
198  const gsl::not_null<int*> iter,
199  const gsl::not_null<bool*> already_visited,
200  Args&&... /*args*/) {
201  if (box.which() == *iter and not*already_visited) {
202  ERROR("Cannot call apply function of '"
203  << pretty_type::get_name<Invokable>() << "' with DataBox type '"
204  << pretty_type::get_name<ThisVariant>() << "' and arguments '"
205  << pretty_type::get_name<tmpl::list<Args...>>() << "'");
206  }
207  (*iter)++;
208 }

210 /*!
211  * \brief Calls an `Invokable`'s `apply` static member function with the current
212  * type in the `boost::variant`.
213  *
214  * The primary use case for this is to allow executing a single Action at any
215  * point in the Algorithm. The current best-known use case for this is setting
216  * up initial data. However, the implementation is generic enough to handle a
217  * call at any time that is valid. Here valid is defined as the `apply` function
218  * only accesses members of the DataBox that are guaranteed to be present when
219  * it is invoked, and returns a DataBox of a type that does not break the
220  * Algorithm.
221  */
222 template <typename Invokable, typename InitialDataBox, typename... Variants,
223  typename... Args>
224 void apply_visitor(boost::variant<Variants...>& box, Args&&... args) {
225  // iter is the current element of the variant in the "for loop"
226  int iter = 0;
227  // already_visited ensures that only one visitor is invoked
228  bool already_visited = false;
229  static_cast<void>(std::initializer_list<char>{
230  (apply_visitor_helper<Invokable, InitialDataBox, Variants>(
231  box, &iter, &already_visited, std::forward<Args>(args)...),
232  '0')...});
233 }

243 /*!
244  * \ingroup ParallelGroup
245  * \brief A distributed object (Charm++ Chare) that executes a series of Actions
246  * and is capable of sending and receiving data. Acts as an interface to
247  * Charm++.
248  *
249  * ### Different Types of Algorithms
250  * Charm++ chares can be one of four types, which is specified by the first
251  * template parameter, `ChareType`. The four available types of Algorithms are:
252  * 1. A Parallel::Algorithms::Singleton where there is only one
253  * in the entire execution of the program.
254  * 2. A Parallel::Algorithms::Array which holds zero or more
255  * elements each of which is a distributed object on some core. An array can
256  * grow and shrink in size dynamically if need be and can also be bound to
257  * another array. That is, the bound array has the same number of elements as
258  * the array it is bound to, and elements with the same ID are on the same core.
259  * 3. A Parallel::Algorithms::Group, which is an array but there is
260  * one element per core and they are not able to be moved around between cores.
261  * These are typically useful for gathering data from array elements on their
262  * core, and then processing or reducing it.
263  * 4. A Parallel::Algorithms::Nodegroup, which is similar to a
264  * group except that there is one element per node. For Charm++ SMP (shared
265  * memory parallelism) builds a node corresponds to the usual definition of a
266  * node on a supercomputer. However, for non-SMP builds nodes and cores are
267  * equivalent. An important difference between groups and nodegroups is that
268  * entry methods (remote calls to functions) are not threadsafe on nodegroups.
269  * It is up to the person writing the Actions that will be executed on the
270  * Nodegroup Algorithm to ensure they are threadsafe.
271  *
272  * ### What is an Algorithm?
273  * An Algorithm is a distributed object, a Charm++ chare, that repeatedly
274  * executes a series of Actions. An Action is a struct that has a `static` apply
275  * function with signature:
276  *
277  * \code
278  * template <typename... DbTags, typename... InboxTags, typename Metavariables,
279  * typename ArrayIndex, typename ActionList>
280  * static auto apply(db::DataBox<tmpl::list<DbTags...>>& box,
281  * tuples::TaggedTuple<InboxTags...>& inboxes,
282  * const ConstGlobalCache<Metavariables>& cache,
283  * const ArrayIndex& array_index,
284  * const TemporalId& temporal_id, const ActionList meta);
285  * \endcode
286  *
287  * Note that any of the arguments can be const or non-const references except
288  * `array_index`, which must be a `const&`.
289  *
290  * ### Explicit instantiations of entry methods
291  * The code in src/Parallel/CharmMain.cpp registers all entry methods, and if
292  * one is not properly registered then a static_assert explains how to have it
293  * be registered. If there is a bug in the implementation and an entry method
294  * isn't being registered or hitting a static_assert then Charm++ will give an
295  * error of the following form:
296  *
297  * \verbatim
298  * registration happened after init Entry point: simple_action(), addr:
299  * 0x555a3d0e2090
300  * ------------- Processor 0 Exiting: Called CmiAbort ------------
301  * Reason: Did you forget to instantiate a templated entry method in a .ci file?
302  * \endverbatim
303  *
304  * If you encounter this issue please file a bug report supplying everything
305  * necessary to reproduce the issue.
306  */

695 template <typename ParallelComponent, typename ChareType,
696  typename Metavariables, typename... ActionsPack, typename ArrayIndex,
697  typename InitialDataBox>
698 template <size_t... Is>
699 constexpr bool
700 AlgorithmImpl<ParallelComponent, ChareType, Metavariables,
701  tmpl::list<ActionsPack...>, ArrayIndex, InitialDataBox>::
702  iterate_over_actions(const std::index_sequence<Is...> /*meta*/) noexcept {
703  bool take_next_action = true;
704  const auto helper = [ this, &take_next_action ](auto iteration) noexcept {
705  constexpr size_t iter = decltype(iteration)::value;
706  if (not(take_next_action and not terminate_ and algorithm_step_ == iter)) {
707  return;
708  }
709  using this_action = tmpl::at_c<actions_list, iter>;
710  using this_databox =
711  tmpl::at_c<databox_types,
712  iter == 0 ? tmpl::size<databox_types>::value - 1 : iter>;
713  this_databox* box{};
714 
715  try {
716  box = &boost::get<this_databox>(box_);
717  } catch (std::exception& e) {
718  ERROR(
719  "\nFailed to retrieve Databox in take_next_action:\nCaught "
720  "exception: '"
721  << e.what() << "'\nDataBox type: '"
722  << pretty_type::get_name<this_databox>() << "'\nIteration: " << iter
723  << "\nAction: '" << pretty_type::get_name<this_action>()
724  << "'\nBoost::Variant id: " << box_.which()
725  << "\nBoost::Variant type is: '" << type_of_current_state(box_)
726  << "'\n\n");
727  }
728 
729  const auto check_if_ready = make_overloader(
730  [this, &box](std::true_type /*has_is_ready*/, auto t) {
731  return decltype(t)::is_ready(
732  static_cast<const this_databox&>(*box),
733  static_cast<const tuples::TaggedTupleTypelist<inbox_tags_list>&>(
734  inboxes_),
735  *const_global_cache_,
736  static_cast<const array_index&>(array_index_));
737  },
738  [](std::false_type /*has_is_ready*/, auto) { return true; });
739 
740  if (not check_if_ready(
741  Algorithm_detail::is_is_ready_callable_t<
742  this_action, this_databox,
743  tuples::TaggedTupleTypelist<inbox_tags_list>,
744  Parallel::ConstGlobalCache<Metavariables>, array_index>{},
745  this_action{})) {
746  take_next_action = false;
747  return;
748  }
749 
750 #ifdef SPECTRE_CHARM_PROJECTIONS
751  traceUserBracketEvent(SPECTRE_CHARM_NON_ACTION_WALLTIME_EVENT_ID,
752  non_action_time_start_, Parallel::wall_time());
753  double start_time = detail::start_trace_action<this_action>();
754 #endif
755  performing_action_ = true;
756  algorithm_step_++;
757  make_overloader(
758  [this](auto& my_box, std::integral_constant<size_t, 1> /*meta*/)
759  SPECTRE_JUST_ALWAYS_INLINE noexcept {
760  std::tie(box_) = this_action::apply(
761  my_box, inboxes_, *const_global_cache_,
762  static_cast<const array_index&>(array_index_), actions_list{},
763  std::add_pointer_t<ParallelComponent>{});
764  },
765  [this](auto& my_box, std::integral_constant<size_t, 2> /*meta*/)
766  SPECTRE_JUST_ALWAYS_INLINE noexcept {
767  std::tie(box_, terminate_) = this_action::apply(
768  my_box, inboxes_, *const_global_cache_,
769  static_cast<const array_index&>(array_index_), actions_list{},
770  std::add_pointer_t<ParallelComponent>{});
771  },
772  [this](auto& my_box, std::integral_constant<size_t, 3> /*meta*/)
773  SPECTRE_JUST_ALWAYS_INLINE noexcept {
774  std::tie(box_, terminate_, algorithm_step_) = this_action::apply(
775  my_box, inboxes_, *const_global_cache_,
776  static_cast<const array_index&>(array_index_), actions_list{},
777  std::add_pointer_t<ParallelComponent>{});
778  })(
779  *box, typename std::tuple_size<decltype(this_action::apply(
780  *box, inboxes_, *const_global_cache_,
781  static_cast<const array_index&>(array_index_), actions_list{},
782  std::add_pointer_t<ParallelComponent>{}))>::type{});
783 
784  performing_action_ = false;
785 #ifdef SPECTRE_CHARM_PROJECTIONS
786  detail::stop_trace_action<this_action>(start_time);
787  non_action_time_start_ = Parallel::wall_time();
788 #endif
789  // Wrap counter if necessary
790  if (algorithm_step_ >= sizeof...(ActionsPack)) {
791  algorithm_step_ = 0;
792  }
793  };
794  // In case of no Actions
795  static_cast<void>(helper);
796  // This is a template for loop for Is
797  static_cast<void>(std::initializer_list<char>{
798  (static_cast<void>(helper(std::integral_constant<size_t, Is>{})),
799  '0')...});
800  return take_next_action;
801 }
