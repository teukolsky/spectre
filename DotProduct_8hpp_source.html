14 /*!
15  * \ingroup TensorGroup
16  * \brief Compute the Euclidean dot product of two vectors or one forms
17  *
18  * \details
19  * Returns \f$A^a B^b \delta_{ab}\f$ for input vectors \f$A^a\f$ and \f$B^b\f$
20  * or \f$A_a B_b \delta^{ab}\f$ for input one forms \f$A_a\f$ and \f$B_b\f$.
21  */
22 template <typename DataType, typename Index>
23 Scalar<DataType> dot_product(
24  const Tensor<DataType, Symmetry<1>, index_list<Index>>& vector_a,
25  const Tensor<DataType, Symmetry<1>, index_list<Index>>& vector_b) noexcept {
26  auto dot_product = make_with_value<Scalar<DataType>>(vector_a, 0.);
27  for (size_t d = 0; d < Index::dim; ++d) {
28  get(dot_product) += vector_a.get(d) * vector_b.get(d);
29  }
30  return dot_product;
31 }

55 /*!
56  * \ingroup TensorGroup
57  * \brief Compute the dot_product of two vectors or one forms
58  *
59  * \details
60  * Returns \f$g_{ab} A^a B^b\f$, where \f$g_{ab}\f$ is the metric,
61  * \f$A^a\f$ is vector_a, and \f$B^b\f$ is vector_b.
62  * Or, returns \f$g^{ab} A_a B_b\f$ when given one forms \f$A_a\f$
63  * and \f$B_b\f$ with an inverse metric \f$g^{ab}\f$.
64  */
65 template <typename DataType, typename Index>
66 Scalar<DataType> dot_product(
67  const Tensor<DataType, Symmetry<1>, index_list<Index>>& vector_a,
68  const Tensor<DataType, Symmetry<1>, index_list<Index>>& vector_b,
69  const Tensor<DataType, Symmetry<1, 1>,
70  index_list<change_index_up_lo<Index>,
71  change_index_up_lo<Index>>>& metric) noexcept {
72  auto dot_product = make_with_value<Scalar<DataType>>(vector_a, 0.);
73  for (size_t a = 0; a < Index::dim; ++a) {
74  for (size_t b = 0; b < Index::dim; ++b) {
75  get(dot_product) += vector_a.get(a) * vector_b.get(b) * metric.get(a, b);
76  }
77  }
78  return dot_product;
79 }
