89 /*!
90  * \ingroup UtilitiesGroup
91  * \brief A checked version of narrow_cast() that ERRORs if the cast changed
92  * the value
93  */
94 template <class T, class U>
95 SPECTRE_ALWAYS_INLINE T narrow(U u) {
96  T t = narrow_cast<T>(u);
97  if (static_cast<U>(t) != u) {
98  ERROR("Failed to cast " << u << " of type " << pretty_type::get_name<U>()
99  << " to type " << pretty_type::get_name<T>());
100  }
101  if (not gsl_detail::is_same_signedness<T, U>::value and
102  ((t < T{}) != (u < U{}))) {
103  ERROR("Failed to cast " << u << " of type " << pretty_type::get_name<U>()
104  << " to type " << pretty_type::get_name<T>());
105  }
106  return t;
107 }

158 /*!
159  * \ingroup UtilitiesGroup
160  * \brief Require a pointer to not be a `nullptr`
161  *
162  * Restricts a pointer or smart pointer to only hold non-null values.
163  *
164  * Has zero size overhead over `T`.
165  *
166  * If `T` is a pointer (i.e. `T == U*`) then
167  * - allow construction from `U*`
168  * - disallow construction from `nullptr_t`
169  * - disallow default construction
170  * - ensure construction from null `U*` fails
171  * - allow implicit conversion to `U*`
172  */
173 template <class T>
174 class not_null {
175  public:
176  static_assert(std::is_assignable<T&, std::nullptr_t>::value,
177  "T cannot be assigned nullptr.");
178 
179  template <typename U, Requires<std::is_convertible<U, T>::value> = nullptr>
180  constexpr not_null(U&& u) : ptr_(std::forward<U>(u)) {
181  Expects(ptr_ != nullptr);
182  }
183 
184  template <typename U, Requires<std::is_convertible<U, T>::value> = nullptr>
185  constexpr not_null(const not_null<U>& other) : not_null(other.get()) {}
186 
187  not_null(const not_null& other) = default;
188  not_null& operator=(const not_null& other) = default;
189 
190  constexpr T get() const {
191  Ensures(ptr_ != nullptr);
192  return ptr_;
193  }
194 
195  constexpr operator T() const { return get(); }
196  constexpr T operator->() const { return get(); }
197  constexpr decltype(auto) operator*() const { return *get(); }
198 
199  // prevents compilation when someone attempts to assign a null pointer
200  // constant
201  not_null(std::nullptr_t) = delete;
202  not_null& operator=(std::nullptr_t) = delete;
203 
204  // unwanted operators...pointers only point to single objects!
205  not_null& operator++() = delete;
206  not_null& operator--() = delete;
207  not_null operator++(int) = delete;
208  not_null operator--(int) = delete;
209  not_null& operator+=(std::ptrdiff_t) = delete;
210  not_null& operator-=(std::ptrdiff_t) = delete;
211  void operator[](std::ptrdiff_t) const = delete;
212 
213  private:
214  T ptr_;
215 };
