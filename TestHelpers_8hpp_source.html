48 /// \ingroup TestingFrameworkGroup
49 /// \brief Test the serialization of a derived class via a base class pointer
50 /// \example
51 /// \snippet Test_PupStlCpp11.cpp example_serialize_derived
52 /// \tparam B the base class
53 /// \tparam D the derived class
54 /// \tparam Args deduced from `args`
55 /// \param args arguments passed to a constructor of the derived class
56 template <typename B, typename D, typename... Args>
57 void test_serialization_via_base(Args&&... args) {
58  static_assert(cpp17::is_base_of_v<B, D>,
59  "passed input type is not derived from specified base");
60  static_assert(tt::has_equivalence_v<D>, "No operator== for derived class");
61  Parallel::register_derived_classes_with_charm<B>();
62  std::unique_ptr<B> base = std::make_unique<D>(args...);
63  std::unique_ptr<B> pupped_base = serialize_and_deserialize(base);
64  CHECK_FALSE(nullptr == dynamic_cast<const D*>(pupped_base.get()));
65  const D derived(args...);
66  CHECK(derived == dynamic_cast<const D&>(*pupped_base));
67 }

69 /// Test for copy semantics assuming operator== is implement correctly
70 template <typename T, Requires<tt::has_equivalence<T>::value> = nullptr>
71 void test_copy_semantics(const T& a) {
72  static_assert(std::is_copy_assignable<T>::value,
73  "Class is not copy assignable.");
74  static_assert(std::is_copy_constructible<T>::value,
75  "Class is not copy constructible.");

77  CHECK(b == a);

81  CHECK(c == a);
82  // clang-tidy: self-assignment

87 /// Test for move semantics assuming operator== is implemented correctly.
88 /// \requires `std::is_rvalue_reference<decltype(a)>::%value` is true.
89 /// If T is not default constructible, you pass additional
90 /// arguments that are used to construct a T.
91 template <typename T, Requires<tt::has_equivalence<T>::value> = nullptr,
92  typename... Args>
93 void test_move_semantics(T&& a, const T& comparison, Args&&... args) {
94  static_assert(std::is_rvalue_reference<decltype(a)>::value,
95  "Must move into test_move_semantics");
96  static_assert(std::is_nothrow_move_assignable<T>::value,
97  "Class is not nothrow move assignable.");
98  static_assert(std::is_nothrow_move_constructible<T>::value,
99  "Class is not nothrow move constructible.");
100  if (&a == &comparison or a != comparison) {
101  // We use ERROR instead of ASSERT (which we normally should be using) to
102  // guard against someone writing tests in Release mode where ASSERTs don't
103  // show up.
104  ERROR("'a' and 'comparison' must be distinct (but equal in value) objects");
105  }
106  T b(std::forward<Args>(args)...);
107  // clang-tidy: use std::forward instead of std::move
108  b = std::move(a); // NOLINT
109  CHECK(b == comparison);
110  T c(std::move(b));
111  CHECK(c == comparison);
112 }

130 // Test for reverse iterators
131 template <typename Container>
132 void test_reverse_iterators(Container& c) {
133  CHECK(std::distance(c.rbegin(), c.rend()) ==
134  static_cast<decltype(std::distance(c.rbegin(), c.rend()))>(c.size()));
135 
136  CHECK(c.rbegin() == c.crbegin());
137  CHECK(c.rend() == c.crend());
138 
139  auto it = c.begin();
140  auto rit = c.rbegin();
141  auto end = c.end();
142  auto rend = c.rend();
143  auto cit = c.cbegin();
144  auto cend = c.cend();
145  auto crit = c.crbegin();
146  auto crend = c.crend();
147 
148  for (size_t i = 0; i < c.size(); i++) {
149  CHECK(*it == *(std::prev(rend, 1)));
150  CHECK(*rit == *(std::prev(end, 1)));
151  CHECK(*cit == *(std::prev(crend, 1)));
152  CHECK(*crit == *(std::prev(cend, 1)));
153  it++;
154  rit++;
155  rend--;
156  end--;
157  crit++;
158  cit++;
159  crend--;
160  cend--;
161  }
162 
163  const auto& const_c = c;
164  CHECK(std::distance(const_c.begin(), const_c.end()) ==
165  static_cast<decltype(std::distance(const_c.begin(), const_c.end()))>(
166  const_c.size()));
167  auto c_it = const_c.begin();
168  auto c_rit = const_c.rbegin();
169  auto c_end = const_c.end();
170  auto c_rend = const_c.rend();
171  for (size_t i = 0; i < c.size(); i++) {
172  CHECK(*c_it == *(std::prev(c_rend, 1)));
173  CHECK(*c_rit == *(std::prev(c_end, 1)));
174  c_it++;
175  c_rit++;
176  c_rend--;
177  c_end--;
178  }
179 }

232 /*!
233  * \ingroup TestingFrameworkGroup
234  * \brief Calculates the derivative of an Invocable at a point x - represented
235  * by an array of doubles - in the domain of `map` with a sixth-order finite
236  * difference method.
237  *
238  * \details Intended for use with CoordinateMaps taking the domain {xi,eta,zeta}
239  * to the range {x,y,z}. This function calculates the derivative along the
240  * direction given by `direction` with a step size of `h`.
241  *
242  * \requires direction be between 0 and VolumeDim
243  */
244 template <typename Invocable, size_t VolumeDim>
245 std::array<double, VolumeDim> numerical_derivative(
246  const Invocable& map, const std::array<double, VolumeDim>& x,
247  const size_t direction, const double delta) {
248  ASSERT(0 <= direction and direction < VolumeDim,
249  "Trying to take derivative along axis " << direction);
250 
251  const auto dx = [direction, delta]() {
252  auto d = make_array<VolumeDim>(0.);
253  gsl::at(d, direction) = delta;
254  return d;
255  }();
256 
257  const std::array<double, VolumeDim> x_1ahead = x + dx;
258  const std::array<double, VolumeDim> x_2ahead = x_1ahead + dx;
259  const std::array<double, VolumeDim> x_3ahead = x_2ahead + dx;
260  const std::array<double, VolumeDim> x_1behind = x - dx;
261  const std::array<double, VolumeDim> x_2behind = x_1behind - dx;
262  const std::array<double, VolumeDim> x_3behind = x_2behind - dx;
263  return (1.0 / (60.0 * delta)) * map(x_3ahead) +
264  (-3.0 / (20.0 * delta)) * map(x_2ahead) +
265  (0.75 / delta) * map(x_1ahead) + (-0.75 / delta) * map(x_1behind) +
266  (3.0 / (20.0 * delta)) * map(x_2behind) +
267  (-1.0 / (60.0 * delta)) * map(x_3behind);
268 }
