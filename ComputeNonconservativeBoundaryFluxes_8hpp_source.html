25 /// \ingroup ActionsGroup
26 /// \ingroup DiscontinuousGalerkinGroup
27 /// \brief Compute \f$\hat{n} \cdot F\f$ on the boundaries for a
28 /// non-conservative system.
29 ///
30 /// Uses:
31 /// - ConstGlobalCache: nothing
32 /// - DataBox:
33 /// Tags::InternalDirections<volume_dim>,
34 /// Tags::Interface<
35 /// Tags::InternalDirections<volume_dim>,
36 /// Tags::Normalized<Tags::UnnormalizedFaceNormal<volume_dim>>>,
37 /// interface items as required by Metavariables::system::normal_dot_fluxes
38 ///
39 /// DataBox changes:
40 /// - Adds: nothing
41 /// - Removes: nothing
42 /// - Modifies:
43 /// Tags::Interface<
44 /// Tags::InternalDirections<volume_dim>,
45 /// db::add_tag_prefix<Tags::NormalDotFlux, variables_tag>>
46 struct ComputeNonconservativeBoundaryFluxes {
47  private:
48  template <typename Metavariables, typename... NormalDotFluxTags, size_t Dim,
49  typename Frame, typename... Args>
50  static void apply_flux(
51  gsl::not_null<Variables<tmpl::list<NormalDotFluxTags...>>*> boundary_flux,
52  const tnsr::i<DataVector, Dim, Frame>& unit_face_normal,
53  const Args&... boundary_variables) noexcept {
54  Metavariables::system::normal_dot_fluxes::apply(
55  make_not_null(&get<NormalDotFluxTags>(*boundary_flux))...,
56  boundary_variables..., unit_face_normal);
57  }
58 
59  public:
60  template <typename DbTags, typename... InboxTags, typename Metavariables,
61  typename ArrayIndex, typename ActionList,
62  typename ParallelComponent>
63  static auto apply(db::DataBox<DbTags>& box,
64  tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
65  Parallel::ConstGlobalCache<Metavariables>& /*cache*/,
66  const ArrayIndex& /*array_index*/,
67  const ActionList /*meta*/,
68  const ParallelComponent* const /*meta*/) noexcept {
69  using system = typename Metavariables::system;
70  using variables_tag = typename system::variables_tag;
71 
72  using internal_directions_tag =
73  Tags::InternalDirections<system::volume_dim>;
74 
75  using normal_tag = Tags::UnnormalizedFaceNormal<system::volume_dim>;
76 
77  using unit_normal_tag =
78  Tags::Interface<internal_directions_tag, Tags::Normalized<normal_tag>>;
79  using interface_normal_dot_fluxes_tag =
80  Tags::Interface<internal_directions_tag,
81  db::add_tag_prefix<Tags::NormalDotFlux, variables_tag>>;
82 
83  db::mutate_apply<
84  tmpl::list<interface_normal_dot_fluxes_tag>,
85  tmpl::push_front<
86  tmpl::transform<
87  typename Metavariables::system::normal_dot_fluxes::
88  argument_tags,
89  tmpl::bind<Tags::Interface, internal_directions_tag, tmpl::_1>>,
90  internal_directions_tag, unit_normal_tag>>(
91  [](const gsl::not_null<db::item_type<interface_normal_dot_fluxes_tag>*>
92  boundary_fluxes,
93  const db::item_type<internal_directions_tag>& internal_directions,
94  const db::item_type<unit_normal_tag>& unit_face_normals,
95  const auto&... tensors) noexcept {
96  for (const auto& direction : internal_directions) {
97  apply_flux<Metavariables>(
98  make_not_null(&boundary_fluxes->at(direction)),
99  unit_face_normals.at(direction), tensors.at(direction)...);
100  }
101  return boundary_fluxes;
102  },
103  make_not_null(&box));
104 
105  return std::forward_as_tuple(std::move(box));
106  }
107 };
108 } // namespace Actions
