149 // @{
150 /// \ingroup ParallelGroup
151 /// \brief Access data in the cache
152 ///
153 /// \requires ConstGlobalCacheTag is a tag in tag_list
154 ///
155 /// \returns a constant reference to an object in the cache
156 template <typename ConstGlobalCacheTag, typename Metavariables>
157 auto get(const ConstGlobalCache<Metavariables>& cache) noexcept -> const
158  ConstGlobalCache_detail::type_for_get<ConstGlobalCacheTag, Metavariables>& {
159  // We check if the tag is to be retrieved directly or via a base class
160  using tag = tmpl::front<ConstGlobalCache_detail::get_list_of_matching_tags<
161  ConstGlobalCacheTag, Metavariables>>;
162  static_assert(tmpl::size<ConstGlobalCache_detail::get_list_of_matching_tags<
163  ConstGlobalCacheTag, Metavariables>>::value == 1,
164  "Found more than one tag matching the ConstGlobalCacheTag "
165  "requesting to be retrieved.");
166  return make_overloader(
167  [](std::true_type /*is_unique_ptr*/, auto&& local_cache)
168  -> decltype(
169  *(tuples::get<tag>(local_cache.const_global_cache_).get())) {
170  return *(
171  tuples::get<tag>(local_cache.const_global_cache_)
172  .get());
173  },
174  [](std::false_type /*is_unique_ptr*/, auto&& local_cache)
175  -> decltype(tuples::get<tag>(local_cache.const_global_cache_)) {
176  return tuples::get<tag>(
177  local_cache.const_global_cache_);
178  })(typename tt::is_a<std::unique_ptr, typename tag::type>::type{}, cache);
179 }
