37

52  typename db::item_type<normal_dot_fluxes_tag>::tags_list,
53  typename PackagedData::tags_list, tmpl::list<MagnitudeOfFaceNormal>>>>;
54 
55  private:

57  struct BasedMortars : Base, Tags::Mortars<Tag, VolumeDim> {};
58 
59  public:
60  /// The DataBox tag for the data stored on the mortars for global
61  /// stepping.
62  using simple_mortar_data_tag =
63  BasedMortars<Tags::VariablesBoundaryData,
64  Tags::SimpleBoundaryData<
65  db::item_type<typename Metavariables::temporal_id>,
66  LocalData, PackagedData>,
67  volume_dim>;
68 
69  /// The inbox tag for flux communication.
70  struct FluxesTag {
71  using temporal_id = db::item_type<typename Metavariables::temporal_id>;
72  using type = std::map<

74  std::unordered_map<

76  std::pair<temporal_id, PackagedData>,
77  boost::hash<
78  std::pair<Direction<volume_dim>, ElementId<volume_dim>>>>>;
79  };
80 };
81 } // namespace dg
