14 /*!
15  * \ingroup GeneralRelativityGroup
16  * \brief Computes the auxiliary variable \f$\Phi_{iab}\f$ used by the
17  * generalized harmonic formulation of Einstein's equations.
18  *
19  * \details If \f$ N, N^i\f$ and \f$ g_{ij} \f$ are the lapse, shift and spatial
20  * metric respectively, then \f$\Phi_{iab} \f$ is computed as
21  *
22  * \f{align}
23  * \Phi_{ktt} &= - 2 N \partial_k N
24  * + 2 g_{mn} N^m \partial_k N^n
25  * + N^m N^n \partial_k g_{mn} \\
26  * \Phi_{kti} &= g_{mi} \partial_k N^m
27  * + N^m \partial_k g_{mi} \\
28  * \Phi_{kij} &= \partial_k g_{ij}
29  * \f}
30  */
31 template <size_t SpatialDim, typename Frame, typename DataType>
32 tnsr::iaa<DataType, SpatialDim, Frame> phi(
33  const Scalar<DataType>& lapse,
34  const tnsr::i<DataType, SpatialDim, Frame>& deriv_lapse,
35  const tnsr::I<DataType, SpatialDim, Frame>& shift,
36  const tnsr::iJ<DataType, SpatialDim, Frame>& deriv_shift,
37  const tnsr::ii<DataType, SpatialDim, Frame>& spatial_metric,
38  const tnsr::ijj<DataType, SpatialDim, Frame>&
39  deriv_spatial_metric) noexcept;

41 /*!
42  * \ingroup GeneralRelativityGroup
43  * \brief Computes the conjugate momentum \f$\Pi_{ab}\f$ of the spacetime metric
44  * \f$ \psi_{ab} \f$.
45  *
46  * \details If \f$ N, N^i\f$ are the lapse and shift
47  * respectively, and \f$ \Phi_{iab} = \partial_i \psi_{ab} \f$ then
48  * \f$\Pi_{\mu\nu} = -(1/N) ( \partial_t \psi_{\mu\nu} -
49  * N^m \Phi_{m\mu\nu}) \f$ where \f$ \partial_t \psi_{ab} \f$ is computed
50  * as
51  *
52  * \f{align}
53  * \partial_t \psi_{tt} &= - 2 N \partial_t N
54  * + 2 g_{mn} N^m \partial_t N^n
55  * + N^m N^n \partial_t g_{mn} \\
56  * \partial_t \psi_{ti} &= g_{mi} \partial_t N^m
57  * + N^m \partial_t g_{mi} \\
58  * \partial_t \psi_{ij} &= \partial_t g_{ij}
59  * \f}
60  */
61 template <size_t SpatialDim, typename Frame, typename DataType>
62 tnsr::aa<DataType, SpatialDim, Frame> pi(
63  const Scalar<DataType>& lapse, const Scalar<DataType>& dt_lapse,
64  const tnsr::I<DataType, SpatialDim, Frame>& shift,
65  const tnsr::I<DataType, SpatialDim, Frame>& dt_shift,
66  const tnsr::ii<DataType, SpatialDim, Frame>& spatial_metric,
67  const tnsr::ii<DataType, SpatialDim, Frame>& dt_spatial_metric,
68  const tnsr::iaa<DataType, SpatialDim, Frame>& phi) noexcept;

70 /*!
71  * \ingroup GeneralRelativityGroup
72  * \brief Computes generalized harmonic gauge source function.
73  * \details If \f$N, N^i, g_{ij}, \Gamma_{ijk}, K\f$ are the lapse, shift,
74  * spatial metric, spatial Christoffel symbols, and trace of the extrinsic
75  * curvature, then we compute
76  * \f{align}
77  * H_l &= N^{-2} g_{il}(\partial_t N^i - N^k \partial_k N^i)
78  * + N^{-1} \partial_l N - g^{km}\Gamma_{lkm} \\
79  * H_0 &= -N^{-1} \partial_t N + N^{-1} N^k\partial_k N + N^k H_k - N K
80  * \f}
81  * See Eqs. 8 and 9 of [arXiv:gr-qc/0512093]
82  * (https://arxiv.org/abs/gr-qc/0512093)
83  */
84 template <size_t SpatialDim, typename Frame, typename DataType>
85 tnsr::a<DataType, SpatialDim, Frame> gauge_source(
86  const Scalar<DataType>& lapse, const Scalar<DataType>& dt_lapse,
87  const tnsr::i<DataType, SpatialDim, Frame>& deriv_lapse,
88  const tnsr::I<DataType, SpatialDim, Frame>& shift,
89  const tnsr::I<DataType, SpatialDim, Frame>& dt_shift,
90  const tnsr::iJ<DataType, SpatialDim, Frame>& deriv_shift,
91  const tnsr::ii<DataType, SpatialDim, Frame>& spatial_metric,
92  const Scalar<DataType>& trace_extrinsic_curvature,

96 /*!
97  * \ingroup GeneralRelativityGroup
98  * \brief Computes extrinsic curvature from generalized harmonic variables
99  * and the spacetime normal vector.
100  *
101  * \details If \f$ \Pi_{ab} \f$ and \f$ \Phi_{iab} \f$ are the generalized
102  * harmonic conjugate momentum and spatial derivative variables, and if
103  * \f$n^a\f$ is the spacetime normal vector, then the extrinsic curvature

106  * K_{ij} &= \frac{1}{2} \Pi_{ij} + \Phi_{(ij)a} n^a
107  * \f}
108  */

120  * \details If \f$ \Phi_{kab} \f$ is the generalized
