33 /// \ingroup SurfacesGroup
34 /// \brief Computes 3-covariant gradient \f$D_i S_j\f$ of a
35 /// Strahlkorper's normal.
36 ///
37 /// \details See Eqs. (1--9) of https://arxiv.org/abs/gr-qc/9606010.
38 /// Here \f$S_j\f$ is the (normalized) unit one-form to the surface,
39 /// and \f$D_i\f$ is the spatial covariant derivative. Note that this
40 /// object is symmetric, even though this is not obvious from the
41 /// definition. The input arguments `r_hat`, `radius`, and
42 /// `d2x_radius` depend on the Strahlkorper but not on the metric, and
43 /// can be computed from a Strahlkorper using ComputeItems in
44 /// `StrahlkorperTags`. The input argument
45 /// `one_over_one_form_magnitude` is \f$1/\sqrt{g^{ij}n_i n_j}\f$,
46 /// where \f$n_i\f$ is `StrahlkorperTags::NormalOneForm` (i.e. the
47 /// unnormalized one-form to the Strahlkorper); it can be computed
48 /// using (one over) the `magnitude` function. The input argument
49 /// `unit_normal_one_form` is \f$S_j\f$,the normalized one-form.
50 template <typename Frame>
51 tnsr::ii<DataVector, 3, Frame> grad_unit_normal_one_form(
52  const tnsr::i<DataVector, 3, Frame>& r_hat, const DataVector& radius,
53  const tnsr::i<DataVector, 3, Frame>& unit_normal_one_form,
54  const tnsr::ii<DataVector, 3, Frame>& d2x_radius,
55  const DataVector& one_over_one_form_magnitude,

58 /// \ingroup SurfacesGroup
59 /// \brief Computes inverse 2-metric \f$g^{ij}-S^i S^j\f$ of a Strahlkorper.
60 ///
61 /// \details See Eqs. (1--9) of https://arxiv.org/abs/gr-qc/9606010.
62 /// Here \f$S^i\f$ is the (normalized) unit vector to the surface,
63 /// and \f$g^{ij}\f$ is the 3-metric. This object is expressed in the
64 /// usual 3-d Cartesian basis, so it is written as a 3-dimensional tensor.
65 /// But because it is orthogonal to \f$S_i\f$, it has only 3 independent

68 /// \f$S^i = g^{ij} S_j\f$, where \f$S_j\f$ is the unit normal one form.

88 /*!
89  * \ingroup SurfacesGroup
90  * \brief Extrinsic curvature of a 2D `Strahlkorper` embedded in a 3D space.
91  *
92  * \details Implements Eq. (D.43) of Carroll's Spacetime and Geometry text.
93  * Specifically,
94  * \f$ K_{ij} = P^k_i P^l_j \nabla_{(k} S_{l)} \f$, where
95  * \f$ P^k_i = \delta^k_i - S^k S_i \f$,
96  * `grad_normal` is the quantity \f$ \nabla_k S_l \f$ returned by
97  * `StrahlkorperGr::grad_unit_normal_one_form`, and `unit_normal_vector` is
98  * \f$S^i = g^{ij} S_j\f$ where \f$S_j\f$ is the unit normal one form.
99  * Not to be confused with the extrinsic curvature of a 3D spatial slice
100  * embedded in 3+1 spacetime.
101  * Because StrahlkorperGr::grad_unit_normal_one_form is symmetric, this
102  * can be expanded into
103  * \f$ K_{ij} = \nabla_{i}S_{j} - 2 S^k S_{(i}\nabla_{j)}S_k
104  * + S_i S_j S^k S^l \nabla_{k} n_{l}\f$.
105  */
106 template <typename Frame>
107 tnsr::ii<DataVector, 3, Frame> extrinsic_curvature(
108  const tnsr::ii<DataVector, 3, Frame>& grad_normal,
109  const tnsr::i<DataVector, 3, Frame>& unit_normal_one_form,
110  const tnsr::I<DataVector, 3, Frame>& unit_normal_vector) noexcept;

112 /// \ingroup SurfacesGroup
113 /// \brief Intrinsic Ricci scalar of a 2D `Strahlkorper`.
114 ///
115 /// \details Implements Eq. (D.51) of
116 /// Sean Carroll's Spacetime and Geometry textbook (except correcting
117 /// sign errors: both extrinsic curvature terms are off by a minus sign
118 /// in Carroll's text but correct in Carroll's errata).
119 /// \f$ \hat{R}=R - 2 R_{ij} S^i S^j + K^2-K^{ij}K_{ij}.\f$
120 /// Here \f$\hat{R}\f$ is the intrinsic Ricci scalar curvature of
121 /// the Strahlkorper, \f$R\f$ and \f$R_{ij}\f$ are the Ricci scalar and
122 /// Ricci tensor of the 3D space that contains the Strahlkorper,
123 /// \f$ K_{ij} \f$ the output of StrahlkorperGr::extrinsic_curvature,
124 /// \f$ K \f$ is the trace of \f$K_{ij}\f$,
125 /// and `unit_normal_vector` is
126 /// \f$S^i = g^{ij} S_j\f$ where \f$S_j\f$ is the unit normal one form.
127 template <typename Frame>
128 Scalar<DataVector> ricci_scalar(
129  const tnsr::ii<DataVector, 3, Frame>& spatial_ricci_tensor,
130  const tnsr::I<DataVector, 3, Frame>& unit_normal_vector,
131  const tnsr::ii<DataVector, 3, Frame>& extrinsic_curvature,
132  const tnsr::II<DataVector, 3, Frame>& upper_spatial_metric) noexcept;

134 /*!
135  * \ingroup SurfacesGroup
136  * \brief Area element of a 2D `Strahlkorper`.
137  *
138  * \details Implements Eq. (D.13), using Eqs. (D.4) and (D.5),
139  * of https://arxiv.org/abs/gr-qc/9606010.
140  * Specifically, computes
141  * \f$\sqrt{(\Theta^i\Theta_i)(\Phi^j\Phi_j)-(\Theta^i\Phi_i)^2}\f$,
142  * \f$\Theta^i=\left(n^i(n_j-s_j) r J^j_\theta + r J^i_\theta\right)\f$,
143  * \f$\Phi^i=\left(n^i(n_j-s_j)r J^j_\phi + r J^i_\phi\right)\f$,
144  * and \f$\Theta^i\f$ and \f$\Phi^i\f$ are lowered by the
145  * 3D spatial metric \f$g_{ij}\f$. Here \f$J^i_\alpha\f$, \f$s_j\f$,
146  * \f$r\f$, and \f$n^i=n_i\f$ correspond to the input arguments
147  * `jacobian`, `normal_one_form`, `radius`, and `r_hat`, respectively;
148  * these input arguments depend only on the Strahlkorper, not on the
149  * metric, and can be computed from a Strahlkorper using ComputeItems
150  * in `StrahlkorperTags`. Note that this does not include the factor
151  * of \f$\sin\theta\f$, i.e., this returns \f$r^2\f$ for flat space.
152  * This choice makes the area element returned here compatible with
153  * `definite_integral` defined in `YlmSpherePack.hpp`.
154  */
155 template <typename Frame>
156 Scalar<DataVector> area_element(
157  const tnsr::ii<DataVector, 3, Frame>& spatial_metric,
158  const StrahlkorperTags::StrahlkorperTags_detail::Jacobian<Frame>& jacobian,
159  const tnsr::i<DataVector, 3, Frame>& normal_one_form,
160  const DataVector& radius,
161  const tnsr::i<DataVector, 3, Frame>& r_hat) noexcept;

163 /*!
164  * \ingroup SurfacesGroup
165  * \brief Spin function of a 2D `Strahlkorper`.
166  *
167  * \details See Eqs. (2) and (10)
168  * of https://arxiv.org/abs/1708.07325. This function computes the
169  * "spin function," which is an ingredient for horizon surface integrals that
170  * measure quasilocal spin. This function is proportional to the imaginary part
171  * of the horizon's complex scalar curvature. For Kerr black holes, the spin
172  * function is proportional to the horizon vorticity. It is also useful for
173  * visualizing the direction of a black hole's spin.
174  * Specifically, this function computes
175  * \f$\Omega = \epsilon^{AB}\nabla_A\omega_B\f$,
176  * where capital indices index the tangent bundle of the surface and
177  * where \f$\omega_\mu=(K_{\rho\nu}-K g_{\rho\nu})h_\mu^\rho s^\nu\f$ is
178  * the curl of the angular momentum density of the surface,
179  * \f$h^\rho_\mu = \delta_\mu^\rho + u_\mu u^\rho - s_\mu s^\rho\f$
180  * is the projector tangent to the 2-surface,
181  * \f$g_{\rho\nu} = \psi_{\rho\nu} + u_\rho u_\nu\f$ is the spatial
182  * metric of the spatial slice, \f$u^\rho\f$ is the unit normal to the
183  * spatial slice, and \f$s^\nu\f$ is the unit normal vector to the surface.
184  * Because the tangent basis vectors \f$e_A^\mu\f$ are
185  * orthogonal to both \f$u^\mu\f$ and \f$s^\mu\f$, it is straightforward
186  * to show that \f$\Omega = \epsilon^{AB} \nabla_A K_{B\mu}s^\mu\f$.
187  * This function uses the tangent vectors of the `Strahlkorper` to
188  * compute \f$K_{B\mu}s^\mu\f$ and then numerically computes the
189  * components of its gradient. The argument `area_element`
190  * can be computed via `StrahlkorperGr::area_element`.
191  * The argument `unit_normal_vector` can be found by raising the
192  * index of the one-form returned by `StrahlkorperGr::unit_normal_oneform`.
193  * The argument `tangents` is a Tangents that can be obtained from the
194  * StrahlkorperDataBox using the "Tangents" tag.
195  */
196 template <typename Frame>
197 Scalar<DataVector> spin_function(
198  const StrahlkorperTags::StrahlkorperTags_detail::Jacobian<Frame>& tangents,
199  const YlmSpherepack& ylm,
200  const tnsr::I<DataVector, 3, Frame>& unit_normal_vector,
201  const Scalar<DataVector>& area_element,
202  const tnsr::ii<DataVector, 3, Frame>& extrinsic_curvature) noexcept;
