16 /*!
17  * \ingroup ParallelGroup
18  * \brief Used for reducing heterogeneous collection of types in a single
19  * reduction call
20  *
21  * For each `ReductionData` you must write a custom reducer to reduce the data.
22  * Here is an example of such a function:
23  *
24  * \snippet Test_AlgorithmReduction.cpp custom_reduce_function
25  *
26  * A variable number of `CkReductionMsg*`s are passed to the reducer, and so it
27  * is necessary to loop over each message, contributing the data into a new
28  * `ReductionData` (called `reduced` in the above example). The
29  * `Parallel::get()` function is used to retrieve elements from the
30  * `ReductionData` in a manner that is completely analogous to retrieving
31  * elements from a `std::tuple` using `std::get`. Once the new `ReductionData`
32  * has been filled, a new `CkReductionMsg*` must be returned using
33  * `Parallel::new_reduction_msg()`.
34  */
35 template <class... Ts>
36 struct ReductionData {
37  template <class Arg0, class... Args,
38  Requires<not cpp17::is_same_v<std::decay_t<Arg0>, ReductionData>> =
39  nullptr>
40  explicit ReductionData(Arg0&& arg0, Args&&... args)
41  : data(std::forward<Arg0>(arg0), std::forward<Args>(args)...) {}
42 
43  ReductionData() = default;
44  ReductionData(const ReductionData& /*rhs*/) = default;
45  ReductionData& operator=(const ReductionData& /*rhs*/) = default;
46  ReductionData(ReductionData&& /*rhs*/) = default;
47  ReductionData& operator=(ReductionData&& /*rhs*/) = default;
48  ~ReductionData() = default;
49 
50  explicit ReductionData(CkReductionMsg* const message) noexcept {
51  PUP::fromMem creator(message->getData());
52  creator | *this;
53  }
54 
55  /// \cond
56  // clang-tidy: non-const reference
57  void pup(PUP::er& p) noexcept { p | data; } // NOLINT
58 
59  std::unique_ptr<char[]> packed() noexcept;
60 
61  size_t size() noexcept;
62 
63  private:
64  // clang-tidy: false positive redundant declaration
65  template <size_t Index, class... Us>
66  friend auto get(ReductionData<Us...>& reduction_data) noexcept // NOLINT
67  -> decltype(std::get<Index>(reduction_data.data));
68  // clang-tidy: false positive redundant declaration
69  template <size_t Index, class... Us>
70  friend auto get( // NOLINT
71  const ReductionData<Us...>& reduction_data) noexcept
72  -> decltype(std::get<Index>(reduction_data.data));
73  // clang-tidy: false positive redundant declaration
74  template <size_t Index, class... Us>
75  friend auto get(ReductionData<Us...>&& reduction_data) noexcept // NOLINT
76  -> decltype(std::get<Index>(reduction_data.data));
77  // clang-tidy: false positive redundant declaration
78  template <size_t Index, class... Us>
79  friend auto get( // NOLINT
80  const ReductionData<Us...>&& reduction_data) noexcept
81  -> decltype(std::get<Index>(reduction_data.data));
82 
83  std::tuple<Ts...> data;
84  /// \endcond
85 };

153 /*!
154  * \ingroup ParallelGroup
155  * \brief Perform a reduction from the `sender_component` (typically your own
156  * parallel component) to the `target_component`, performing the `Action` upon
157  * receiving the reduction.
158  *
159  * A Charm++ reducer specifying what type of reduction is to be done must also
160  * be passed (see
161  * [here](http://charm.cs.illinois.edu/manuals/html/charm++/manual.html)).
162  *
163  * \example
164  * Built-in Charm++ reductions are supported as:
165  * \snippet Test_AlgorithmReduction.cpp contribute_to_reduction_example
166  */
167 template <
168  class Action, class ReductionType, class SenderProxy, class TargetProxy,
169  Requires<not Parallel_detail::is_custom_reduction_type_v<ReductionType>> =
170  nullptr>
171 void contribute_to_reduction(const ReductionType& reduction_data,
172  const SenderProxy& sender_component,
173  const TargetProxy& target_component,
174  CkReduction::reducerType reducer) noexcept {
175  CkCallback callback(
176  TargetProxy::index_t::template redn_wrapper_reduction_action<
177  Action, std::decay_t<ReductionType>>(nullptr),
178  target_component);
179  sender_component.ckLocal()->contribute(sizeof(reduction_data),
180  &reduction_data, reducer, callback);
181 }

183 /*!
184  * \ingroup ParallelGroup
185  * \brief Perform a reduction from the `sender_component` (typically your own
186  * parallel component) to the `target_component`, performing the `Action` upon
187  * receiving the reduction.
188  *
189  * The template parameter `F` is a function pointer to the custom reduction
190  * function.
191  *
192  * \example
193  * Let's say you want to perform a custom reduction on an
194  * `int`, `std::unordered_map<std::string, int>`, and `std::vector<int>`. You
195  * would then write a custom reduction function like the following:
196  * \snippet Test_AlgorithmReduction.cpp custom_reduce_function
197  * To have an array element contribute to the reduction you must call the
198  * `contribute_to_reduction` as follows:
199  * \snippet Test_AlgorithmReduction.cpp custom_contribute_to_reduction_example
200  *
201  * \note Registration of the reduction function with Charm++ will be handled
202  * automatically so ignore that portion of the Charm++ manual.
203  */
204 template <Parallel::charmxx::ReducerFunctions F, class Action,
205  class ReductionType, class SenderProxy, class TargetProxy>
206 void contribute_to_reduction(ReductionType&& reduction_data,
207  const SenderProxy& sender_component,
208  const TargetProxy& target_component) noexcept {
209  (void)Parallel::charmxx::RegisterReducerFunction<F>::registrar;
210  CkCallback callback(
211  TargetProxy::index_t::template redn_wrapper_reduction_action<
212  Action, std::decay_t<ReductionType>>(nullptr),
213  target_component);
214  sender_component.ckLocal()->contribute(
215  static_cast<int>(reduction_data.size()), reduction_data.packed().get(),
216  Parallel::charmxx::charm_reducer_functions.at(
217  std::hash<Parallel::charmxx::ReducerFunctions>{}(F)),
218  callback);
219 }
