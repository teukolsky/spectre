68 template <class Metavariables, class ActionList>
69 void DgElementArray<Metavariables, ActionList>::initialize(
70  Parallel::CProxy_ConstGlobalCache<Metavariables>& global_cache,
71  const std::unique_ptr<DomainCreator<volume_dim, Frame::Inertial>>
72  domain_creator,
73  const double initial_time, const double initial_dt) noexcept {
74  auto& dg_element_array = Parallel::get_parallel_component<DgElementArray>(
75  *(global_cache.ckLocalBranch()));
76 
77  auto domain = domain_creator->create_domain();
78  for (const auto& block : domain.blocks()) {
79  const auto initial_ref_levs =
80  domain_creator->initial_refinement_levels()[block.id()];
81  const std::vector<ElementId<volume_dim>> element_ids =
82  initial_element_ids(block.id(), initial_ref_levs);
83  int which_proc = 0;
84  const int number_of_procs = Parallel::number_of_procs();
85  for (size_t i = 0; i < element_ids.size(); ++i) {
86  dg_element_array(ElementIndex<volume_dim>(element_ids[i]))
87  .insert(global_cache, which_proc);
88  which_proc = which_proc + 1 == number_of_procs ? 0 : which_proc + 1;
89  }
90  }
91  dg_element_array.doneInserting();
92 
93  // Set time and DeltaT allowing for integration backwards in time
94  const bool time_reversed = initial_dt < 0;
95  const Slab slab =
96  time_reversed ? Slab::with_duration_to_end(initial_time, -initial_dt)
97  : Slab::with_duration_from_start(initial_time, initial_dt);
98  const Time time = time_reversed ? slab.end() : slab.start();
99  const TimeDelta dt = time_reversed ? -slab.duration() : slab.duration();
100 
101  Parallel::simple_action<dg::Actions::InitializeElement<volume_dim>>(
102  dg_element_array, domain_creator->initial_extents(), std::move(domain),
103  time, dt);
104 }
