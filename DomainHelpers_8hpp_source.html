212 /// \ingroup ComputationalDomainGroup
213 /// Iterates over the corners of a VolumeDim-dimensional cube.
214 template <size_t VolumeDim>
215 class VolumeCornerIterator {
216  public:
217  VolumeCornerIterator() noexcept { setup_from_local_corner_number(); }
218  explicit VolumeCornerIterator(size_t initial_local_corner_number) noexcept
219  : local_corner_number_(initial_local_corner_number) {
220  setup_from_local_corner_number();
221  }
222  VolumeCornerIterator(
223  // The block index is also global corner
224  // index of the lowest corner of the block.
225  Index<VolumeDim> block_index,
226  Index<VolumeDim> global_corner_extents) noexcept
227  : global_corner_number_(
228  collapsed_index(block_index, global_corner_extents)),
229  global_corner_index_(block_index),
230  global_corner_extents_(global_corner_extents) {}
231  void operator++() noexcept {
232  ++local_corner_number_;
233  setup_from_local_corner_number();
234  }
235  explicit operator bool() const noexcept {
236  return local_corner_number_ < two_to_the(VolumeDim);
237  }
238  const size_t& local_corner_number() const noexcept {
239  return local_corner_number_;
240  }
241  size_t global_corner_number() const noexcept {
242  std::array<size_t, VolumeDim> new_indices{};
243  for (size_t i = 0; i < VolumeDim; i++) {
244  gsl::at(new_indices, i) =
245  global_corner_index_[i] +
246  (gsl::at(array_sides_, i) == Side::Upper ? 1 : 0);
247  }
248  const Index<VolumeDim> interior_multi_index(new_indices);
249  return collapsed_index(interior_multi_index, global_corner_extents_);
250  }
251  const std::array<Side, VolumeDim>& operator()() const noexcept {
252  return array_sides_;
253  }
254  const std::array<Side, VolumeDim>& operator*() const noexcept {
255  return array_sides_;
256  }
257  const std::array<double, VolumeDim>& coords_of_corner() const noexcept {
258  return coords_of_corner_;
259  }
260  const std::array<Direction<VolumeDim>, VolumeDim>& directions_of_corner()
261  const noexcept {
262  return array_directions_;
263  }
264  void setup_from_local_corner_number() noexcept {
265  for (size_t i = 0; i < VolumeDim; i++) {
266  gsl::at(coords_of_corner_, i) =
267  2.0 * get_nth_bit(local_corner_number_, i) - 1.0;
268  gsl::at(array_sides_, i) =
269  2 * get_nth_bit(local_corner_number_, i) - 1 == 1 ? Side::Upper
270  : Side::Lower;
271  gsl::at(array_directions_, i) =
272  Direction<VolumeDim>(i, gsl::at(array_sides_, i));
273  }
274  }
275 
276  private:
277  size_t local_corner_number_ = 0;
278  size_t global_corner_number_{std::numeric_limits<size_t>::max()};
279  Index<VolumeDim> global_corner_index_{};
280  Index<VolumeDim> global_corner_extents_{};
281  std::array<Side, VolumeDim> array_sides_ = make_array<VolumeDim>(Side::Lower);
282  std::array<Direction<VolumeDim>, VolumeDim> array_directions_{};
283  std::array<double, VolumeDim> coords_of_corner_ = make_array<VolumeDim>(-1.0);
284 };

286 /// \ingroup ComputationalDomainGroup
287 /// Iterates over the 2^(VolumeDim-1) logical corners of the face of a
288 /// VolumeDim-dimensional cube in the given direction.
289 template <size_t VolumeDim>
290 class FaceCornerIterator {
291  public:
292  explicit FaceCornerIterator(Direction<VolumeDim> direction) noexcept;
293  void operator++() noexcept {
294  face_index_++;
295  do {
296  index_++;
297  } while (get_nth_bit(index_, direction_.dimension()) ==
298  (direction_.side() == Side::Upper ? 0 : 1));
299  for (size_t i = 0; i < VolumeDim; ++i) {
300  corner_[i] = 2 * static_cast<int>(get_nth_bit(index_, i)) - 1;
301  }
302  }
303  explicit operator bool() const noexcept {
304  return face_index_ < two_to_the(VolumeDim - 1);
305  }
306  tnsr::I<double, VolumeDim, Frame::Logical> operator()() const noexcept {
307  return corner_;
308  }
309  tnsr::I<double, VolumeDim, Frame::Logical> operator*() const noexcept {
310  return corner_;
311  }
312 
313  // Returns the value used to construct the logical corner.
314  size_t volume_index() const noexcept { return index_; }
315  // Returns the number of times operator++ has been called.
316  size_t face_index() const noexcept { return face_index_; }
317 
318  private:
319  const Direction<VolumeDim> direction_;
320  size_t index_;
321  size_t face_index_ = 0;
322  tnsr::I<double, VolumeDim, Frame::Logical> corner_;
323 };
