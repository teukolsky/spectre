122 /// \ingroup TestingFrameworkGroup
123 /// A class that mocks the infrastructure needed to run actions. It
124 /// simulates message passing using the inbox infrastructure and
125 /// handles most of the arguments to the apply and is_ready action
126 /// methods.
127 template <typename Metavariables>
128 class ActionRunner {
129  public:
130  // No moving, since MockProxy holds a pointer to us.
131  ActionRunner(const ActionRunner&) = delete;
132  ActionRunner(ActionRunner&&) = delete;
133  ActionRunner& operator=(const ActionRunner&) = delete;
134  ActionRunner& operator=(ActionRunner&&) = delete;
135  ~ActionRunner() = default;
136 
137  using GlobalCache = Parallel::ConstGlobalCache<Metavariables>;
138  using CacheTuple =
139  tuples::TaggedTupleTypelist<typename GlobalCache::tag_list>;
140 
141  /// Construct from the tuple of ConstGlobalCache objects.
142  explicit ActionRunner(CacheTuple cache_contents)
143  : cache_(std::move(cache_contents)) {
144  tmpl::for_each<typename Metavariables::component_list>(
145  [this](auto component) {
146  using Component = tmpl::type_from<decltype(component)>;
147  Parallel::get_parallel_component<Component>(cache_)
148  .set_data(
149  &tuples::get<LocalAlgorithms<Component>>(local_algorithms_),
150  &tuples::get<Inboxes<Component>>(inboxes_));
151  });
152  }
153 
154  /// Call Action::apply as if on the portion of Component labeled by
155  /// array_index.
156  //@{
157  template <typename Component, typename Action, typename DbTags,
158  typename... Args>
159  decltype(auto) apply(db::DataBox<DbTags>& box,
160  const typename Component::index& array_index,
161  Args&&... args) noexcept {
162  return Action::apply(box, inboxes<Component>()[array_index], cache_,
163  array_index, typename Component::action_list{},
164  std::add_pointer_t<Component>{nullptr},
165  std::forward<Args>(args)...);
166  }
167 
168  template <typename Component, typename Action, typename DbTags,
169  typename... Args>
170  decltype(auto) apply(const db::DataBox<DbTags>& box,
171  const typename Component::index& array_index,
172  Args&&... args) noexcept {
173  return Action::apply(box, inboxes<Component>()[array_index], cache_,
174  array_index, typename Component::action_list{},
175  std::add_pointer_t<Component>{nullptr},
176  std::forward<Args>(args)...);
177  }
178  //@}
179 
180  /// Call Action::is_ready as if on the portion of Component labeled
181  /// by array_index.
182  template <typename Component, typename Action, typename DbTags>
183  bool is_ready(const db::DataBox<DbTags>& box,
184  const typename Component::index& array_index) noexcept {
185  return Action::is_ready(box,
186  cpp17::as_const(inboxes<Component>()[array_index]),
187  cache_, array_index);
188  }

190  /// Access the inboxes for a given component.
191  template <typename Component>

199  /// Find the set of array indices on Component where the specified
200  /// inbox is not empty.
201  template <typename Component, typename InboxTag>
202  std::unordered_set<typename Component::index>
203  nonempty_inboxes() noexcept {
204  std::unordered_set<typename Component::index> result;
205  for (const auto& element_box : inboxes<Component>()) {
206  if (not tuples::get<InboxTag>(element_box.second).empty()) {
207  result.insert(element_box.first);
208  }
209  }
210  return result;
211  }
