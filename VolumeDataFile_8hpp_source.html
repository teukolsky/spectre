137 template <size_t Dim, typename Fr>
138 void VolumeFile::write_element_coordinates(
139  const tnsr::I<DataVector, Dim, Fr>& grid_coordinates,
140  const Index<Dim>& extents, const std::string& element_id,
141  const std::vector<std::string>& coordinate_names) {
142  h5::detail::OpenGroup group(file_id_, "/" + element_id,
143  h5::AccessType::ReadWrite);
144  for (size_t d = 0; d < Dim; ++d) {
145  h5::write_data(group.id(), grid_coordinates.get(d), extents,
146  coordinate_names[d]);
147  }
148  std::stringstream ss;
149  ss << " <Geometry Type=\"X";
150  if (Dim > 1) {
151  ss << "_Y";
152  }
153  if (Dim > 2) {
154  ss << "_Z";
155  }
156  ss << "\">\n";
157  std::stringstream extents_in_dims;
158  extents_in_dims << extents[0];
159  for (size_t k = 1; k < Dim; ++k) {
160  extents_in_dims << " " << extents[k];
161  }
162  for (size_t d = 0; d < Dim; ++d) {
163  ss << " <DataItem Dimensions=\"" << extents_in_dims.str()
164  << "\" NumberType=\"Double\" Precision=\"" << precision_
165  << "\" Format=\"HDF5\">\n"
166  << " " << h5_file_name_ << ":/" << element_id << "/"
167  << coordinate_names[d] << "\n </DataItem>\n";
168  }
169  ss << " </Geometry>\n";
170  std::fprintf(xml_, "%s", ss.str().c_str()); // NOLINT
171 }
