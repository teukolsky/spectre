76 /// `Jacobian(i,0)` is \f$\frac{1}{r}\partial x^i/\partial\theta\f$,

186 /// `NormalOneForm(i)` is \f$s_i\f$, the (unnormalized) normal one-form
187 /// to the surface, expressed in Cartesian components.
188 /// This is computed by \f$x_i/r-\partial r_{\rm surf}/\partial x^i\f$,
189 /// where \f$x_i/r\f$ is `Rhat` and
190 /// \f$\partial r_{\rm surf}/\partial x^i\f$ is `DxRadius`.
191 /// See Eq. (8) of [arXiv:gr-qc/9606010] (https://arxiv.org/abs/gr-qc/9606010).
192 /// Note on the word "normal": \f$s_i\f$ points in the correct direction
193 /// (it is "normal" to the surface), but it does not have unit length
194 /// (it is not "normalized"; normalization requires a metric).
195 template <typename Frame>
196 struct NormalOneForm : db::ComputeTag {
197  static std::string name() noexcept { return "NormalOneForm"; }
198  static StrahlkorperTags_detail::OneForm<Frame> function(
199  const db::item_type<DxRadius<Frame>>& dx_radius,
200  const db::item_type<Rhat<Frame>>& r_hat) noexcept;
201  using argument_tags = tmpl::list<DxRadius<Frame>, Rhat<Frame>>;
202 };

204 /// `Tangents(i,j)` is \f$\partial x_{\rm surf}^i/\partial q^j\f$,
205 /// where \f$x_{\rm surf}^i\f$ are the Cartesian coordinates of the
206 /// surface (i.e. `CartesianCoords`) and are considered functions of
207 /// \f$(\theta,\phi)\f$.
208 ///
209 /// \f$\partial/\partial q^0\f$ means
210 /// \f$\partial/\partial\theta\f$; and \f$\partial/\partial q^1\f$
211 /// means \f$\csc\theta\,\,\partial/\partial\phi\f$. Note that the
212 /// vectors `Tangents(i,0)` and `Tangents(i,1)` are orthogonal to the
213 /// `NormalOneForm` \f$s_i\f$, i.e.
214 /// \f$s_i \partial x_{\rm surf}^i/\partial q^j = 0\f$; this statement
215 /// is independent of a metric. Also, `Tangents(i,0)` and
216 /// `Tangents(i,1)` are not necessarily orthogonal to each other,
217 /// since orthogonality between 2 vectors (as opposed to a vector and
218 /// a one-form) is metric-dependent.
219 template <typename Frame>
220 struct Tangents : db::ComputeTag {
221  static std::string name() noexcept { return "Tangents"; }
222  static StrahlkorperTags_detail::Jacobian<Frame> function(
223  const ::Strahlkorper<Frame>& strahlkorper, const DataVector& radius,
224  const db::item_type<Rhat<Frame>>& r_hat,
225  const db::item_type<Jacobian<Frame>>& jac) noexcept;
226  using argument_tags = tmpl::list<Strahlkorper<Frame>, Radius<Frame>,
227  Rhat<Frame>, Jacobian<Frame>>;
228 };
