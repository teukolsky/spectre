21 /// \ingroup ComputationalDomainGroup
22 /// A SegmentId labels a segment of the interval [-1,1].
23 /// A SegmentId is used to (partially) identify an Element.
24 /// The segment is the interval [-1 + 2*(N/D), -1 + 2*(N+1)/D]
25 /// where N = index_ and D = 2 ^ refinement_level_.

83 inline SegmentId SegmentId::id_of_parent() const noexcept {
84  ASSERT(0 != refinement_level_,
85  "Cannot call id_of_parent() on root refinement level!");
86  // The parent has half as many segments as the child.
87  return {refinement_level_ - 1, index_ / 2};
88 }
89 
90 inline SegmentId SegmentId::id_of_child(Side side) const noexcept {
91  // We cannot ASSERT on the maximum level because it's only known at runtime
92  // and only known elsewhere in the code, not by SegmentId. I.e. SegmentId is
93  // too low-level to know about this.
94  // The child has twice as many segments as the parent, so for a particular
95  // parent segment, there is both an upper and lower child segment.
96  if (Side::Lower == side) {
97  return {refinement_level_ + 1, index_ * 2};
98  }
99  return {refinement_level_ + 1, 1 + index_ * 2};
100 }

102 inline SegmentId SegmentId::id_if_flipped() const noexcept {
103  return {refinement_level_, two_to_the(refinement_level_) - 1 - index_};
104 }

106 inline double SegmentId::endpoint(Side side) const noexcept {
107  if (Side::Lower == side) {
108  return -1.0 + (2.0 * index_) / two_to_the(refinement_level_);
109  }
110  return -1.0 + (2.0 * index_ + 2.0) / two_to_the(refinement_level_);
111 }

113 inline bool SegmentId::overlaps(const SegmentId& other) const noexcept {
114  const size_t this_denom = two_to_the(refinement_level_);
115  const size_t other_denom = two_to_the(other.refinement_level_);
116  return index_ * other_denom < (other.index_ + 1) * this_denom and
117  other.index_ * this_denom < (index_ + 1) * other_denom;
118 }
