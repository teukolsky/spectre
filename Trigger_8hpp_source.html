53  template <typename DbTags>
54  bool is_triggered(const db::DataBox<DbTags>& box) noexcept {
55  return call_with_dynamic_type<bool, creatable_classes>(
56  this,
57  [&box](auto* const trigger) noexcept {
58  using TriggerType = std::decay_t<decltype(*trigger)>;
59  return db::apply<typename TriggerType::argument_tags>(
60  *trigger, box);
