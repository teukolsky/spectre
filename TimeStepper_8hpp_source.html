74  /// \brief Compute the change in a boundary quantity due to the
75  /// coupling on the interface.
76  ///
77  /// The coupling function `coupling` should take the local and
78  /// remote flux data and compute the derivative of the boundary
79  /// quantity. These values may be used to form a linear combination
80  /// internally, so the result should have appropriate mathematical
81  /// operators defined to allow that.
82  template <typename LocalVars, typename RemoteVars, typename Coupling>
83  std::result_of_t<const Coupling&(LocalVars, RemoteVars)>
84  compute_boundary_delta(
85  const Coupling& coupling,
86  const gsl::not_null<TimeSteppers::BoundaryHistory<
87  LocalVars, RemoteVars,
88  std::result_of_t<const Coupling&(LocalVars, RemoteVars)>>*>
89  history,
90  const TimeDelta& time_step) const noexcept {
91  return TimeStepper_detail::fake_virtual_compute_boundary_delta<
92  creatable_classes>(this, coupling, history, time_step);
93  }
