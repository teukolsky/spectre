22 /*!
23  * \ingroup SpectralGroup
24  * \brief Functionality associated with a particular choice of basis functions

27  * \details The functions in this namespace provide low-level access to
28  * collocation points, quadrature weights and associated matrices, such as for

47 /*!
48  * \ingroup SpectralGroup
49  * \brief The choice of quadrature method to compute integration weights.
50  *
51  * \details Integrals using \f$N\f$ collocation points with Gauss quadrature are
52  * exact to polynomial order \f$p=2N-1\f$. Gauss-Lobatto quadrature is exact
53  * only to polynomial order \f$p=2N-3\f$, but includes collocation points at the
54  * domain boundary.
55  */
56 enum class Quadrature { Gauss, GaussLobatto };

58 /*!
59  * \brief Minimum number of possible collocation points for a quadrature type.
60  *
61  * \details Since Gauss-Lobatto quadrature has points on the domain boundaries
62  * it must have at least two collocation points. Gauss quadrature can have only
63  * one collocation point.
64  */
65 template <Basis, Quadrature>
66 constexpr size_t minimum_number_of_points{std::numeric_limits<size_t>::max()};
67 /// \cond

72  2;

109 /*!
110  * \brief %Matrix used to compute the derivative of a function.
111  *
112  * \details For a function represented by the nodal coefficients \f$u_j\f$ a
113  * matrix multiplication with the differentiation matrix \f$D_{ij}\f$ gives the
114  * coefficients of the function's derivative. Since \f$u(x)\f$ is expanded in
115  * Lagrange polynomials \f$u(x)=u_j l_j(x)\f$ the differentiation matrix is
116  * computed as \f$D_{ij}=l_j^\prime(\xi_i)\f$ where the \f$\xi_i\f$ are the
117  * collocation points.
118  *
119  * \param num_points The number of collocation points
120  */
121 template <Basis BasisType, Quadrature QuadratureType>
122 const Matrix& differentiation_matrix(size_t num_points) noexcept;

153 /*!
154  * \brief %Matrix used to transform from the spectral coefficients (modes) of a
155  * function to its nodal coefficients. Also referred to as the _Vandermonde
156  * matrix_.
157  *
158  * \details The Vandermonde matrix is computed as
159  * \f$\mathcal{V}_{ij}=\Phi_j(\xi_i)\f$ where the \f$\Phi_j(x)\f$ are the
160  * spectral basis functions used in the modal expansion
161  * \f$u(x)=\widetilde{u}_j\Phi_j(x)\f$, e.g. normalized Legendre polynomials,
162  * and the \f$\xi_i\f$ are the collocation points used to construct the
163  * interpolating Lagrange polynomials in the nodal expansion
164  * \f$u(x)=u_j l_j(x)\f$. Then the Vandermonde matrix arises as
165  * \f$u(\xi_i)=u_i=\widetilde{u}_j\Phi_j(\xi_i)=
166  * \mathcal{V}_{ij}\widetilde{u}_j\f$.
167  *
168  * \param num_points The number of collocation points
169 
170  * \see grid_points_to_spectral_matrix(size_t)
171  */
172 template <Basis BasisType, Quadrature QuadratureType>
173 const Matrix& spectral_to_grid_points_matrix(size_t num_points) noexcept;
174

183 /*!
184  * \brief %Matrix used to transform from the nodal coefficients of a function to
185  * its spectral coefficients (modes). Also referred to as the inverse
186  * _Vandermonde matrix_.
187  *
188  * \details This is the inverse to the Vandermonde matrix \f$\mathcal{V}\f$
189  * computed in spectral_to_grid_points_matrix(size_t). It can be computed
190  * analytically for Gauss quadrature by evaluating
191  * \f$\mathcal{V}^{-1}_{ij}u_j=\widetilde{u}_i=\frac{(u,\Phi_i)}{\gamma_i}\f$
192  * for a Lagrange basis function \f$u(x)=l_k(x)\f$ to find
193  * \f$\mathcal{V}^{-1}_{ij}=\mathcal{V}_{ji}\frac{w_j}{\gamma_i}\f$ where the
194  * \f$w_j\f$ are the Gauss quadrature weights and \f$\gamma_i\f$ is the norm
195  * square of the spectral basis function \f$\Phi_i\f$.
196  *
197  * \param num_points The number of collocation points
198  *
199  * \see spectral_to_grid_points_matrix(size_t)
200  */
201 template <Basis BasisType, Quadrature QuadratureType>
202 const Matrix& grid_points_to_spectral_matrix(size_t num_points) noexcept;
