11  * \ingroup ParallelGroup
12  * \brief Send the data `args...` to the algorithm running on `proxy`, and tag
13  * the message with the identifier `temporal_id`.
14  *
15  * If the algorithm was previously disabled, set `enable_if_disabled` to true to
16  * enable the algorithm on the parallel component.
17  *
18  * \note The reason there are two separate functions is because Charm++ does not
19  * allow defaulted arguments for group and nodegroup chares.
20  */
21 template <typename ReceiveTag, typename Proxy, typename ReceiveDataType>
22 void receive_data(Proxy&& proxy, typename ReceiveTag::temporal_id temporal_id,
23  ReceiveDataType&& receive_data,
24  const bool enable_if_disabled) noexcept {
25  proxy.template receive_data<ReceiveTag>(
26  std::move(temporal_id), std::forward<ReceiveDataType>(receive_data),
27  enable_if_disabled);
28 }

49 void simple_action(Proxy&& proxy, Arg0&& arg0, Args&&... args) noexcept {
50  proxy.template simple_action<Action>(
51  std::tuple<std::decay_t<Arg0>, std::decay_t<Args>...>(
