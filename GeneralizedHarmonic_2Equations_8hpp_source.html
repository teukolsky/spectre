85 /*!
86  * \brief Compute the fluxes of the Generalized Harmonic formulation of
87  * Einstein's equations.
88  *
89  * \details The expressions for the fluxes is obtained
90  * from <a href="https://arxiv.org/abs/gr-qc/0512093">
91  * gr-qc/0512093 </a>.
92  * The fluxes for each variable are obtained by taking the principal part of
93  * equations 35, 36, and 37, and replacing derivatives \f$ \partial_k \f$
94  * with the unit normal \f$ n_k \f$. This gives:
95  *
96  * \f{align*}
97  * F(\psi_{ab}) =& -(1 + \gamma_1) N^k n_k \psi_{ab} \\
98  * F(\Pi_{ab}) =& - N^k n_k \Pi_{ab} + N g^{ki}n_k \Phi_{iab} - \gamma_1
99  * \gamma_2
100  * N^k n_k \psi_{ab} \\
101  * F(\Phi_{iab}) =& - N^k n_k \Phi_{iab} + N n_i \Pi_{ab} - \gamma_1 \gamma_2
102  * N^i \Phi_{iab}
103  * \f}
104  *
105  * where \f$\psi_{ab}\f$ is the spacetime metric, \f$\Pi_{ab}\f$ its conjugate
106  * momentum, \f$ \Phi_{iab} \f$ is an auxiliary field as defined by the tag Phi,
107  * \f$N\f$ is the lapse, \f$ N^k \f$ is the shift, \f$ g^{ki} \f$ is the inverse
108  * spatial metric, and \f$ \gamma_1, \gamma_2 \f$ are constraint damping
109  * parameters.
110  */
111 template <size_t Dim>
112 struct ComputeNormalDotFluxes {
113  public:
114  using argument_tags =
115  tmpl::list<gr::Tags::SpacetimeMetric<Dim>, Pi<Dim>, Phi<Dim>,
116  ConstraintGamma1, ConstraintGamma2, gr::Tags::Lapse<Dim>,
117  gr::Tags::Shift<Dim>, gr::Tags::InverseSpatialMetric<Dim>>;
118 
119  static void apply(
120  gsl::not_null<tnsr::aa<DataVector, Dim>*>
121  spacetime_metric_normal_dot_flux,
122  gsl::not_null<tnsr::aa<DataVector, Dim>*> pi_normal_dot_flux,
123  gsl::not_null<tnsr::iaa<DataVector, Dim>*> phi_normal_dot_flux,
124  const tnsr::aa<DataVector, Dim>& spacetime_metric,
125  const tnsr::aa<DataVector, Dim>& pi,
126  const tnsr::iaa<DataVector, Dim>& phi, const Scalar<DataVector>& gamma1,
127  const Scalar<DataVector>& gamma2, const Scalar<DataVector>& lapse,
128  const tnsr::I<DataVector, Dim>& shift,
129  const tnsr::II<DataVector, Dim>& inverse_spatial_metric,
130  const tnsr::i<DataVector, Dim>& unit_normal) noexcept;
