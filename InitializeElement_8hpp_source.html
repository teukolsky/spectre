57 /// \brief Initialize a dG element with analytic initial data
58 ///
59 /// Uses:
60 /// - ConstGlobalCache:
61 /// * A tag deriving off of Cache::AnalyticSolutionBase

63 ///
64 /// DataBox changes:
65 /// - Adds:
66 /// * Tags::TimeId

73 /// * System::variables_tag

75 /// db::add_tag_prefix<Tags::dt, System::variables_tag>>

91  struct DomainTags {
92  using simple_tags =
93  db::AddSimpleTags<Tags::Mesh<Dim>, Tags::Element<Dim>,
94  Tags::ElementMap<Dim>>;
95 
96  using compute_tags = db::AddComputeTags<
97  Tags::LogicalCoordinates<Dim>,
98  Tags::Coordinates<Tags::ElementMap<Dim>, Tags::LogicalCoordinates<Dim>>,
99  Tags::InverseJacobian<Tags::ElementMap<Dim>,
100  Tags::LogicalCoordinates<Dim>>>;
101 
102  template <typename TagsList>
103  static auto initialize(
104  db::DataBox<TagsList>&& box, const ElementIndex<Dim>& array_index,
105  const std::vector<std::array<size_t, Dim>>& initial_extents,
106  const Domain<Dim, Frame::Inertial>& domain) noexcept {
107  const ElementId<Dim> element_id{array_index};
108  const auto& my_block = domain.blocks()[element_id.block_id()];
109  Mesh<Dim> mesh{initial_extents[element_id.block_id()],
110  Spectral::Basis::Legendre,
111  Spectral::Quadrature::GaussLobatto};
112  Element<Dim> element = create_initial_element(element_id, my_block);
113  ElementMap<Dim, Frame::Inertial> map{
114  element_id, my_block.coordinate_map().get_clone()};
115 
116  return db::create_from<db::RemoveTags<>, simple_tags, compute_tags>(
117  std::move(box), std::move(mesh), std::move(element), std::move(map));
118  }

254  template <typename TagsList, typename Metavariables>
255  static auto initialize(
256  db::DataBox<TagsList>&& box,
257  const Parallel::ConstGlobalCache<Metavariables>& cache,
258  const Time& initial_time, const TimeDelta& initial_dt) noexcept {
259  using Vars = typename variables_tag::type;
260  using DtVars = typename dt_variables_tag::type;
261 
262  const auto& time_stepper = Parallel::get<CacheTags::TimeStepper>(cache);
263 
264  const TimeId time_id(initial_dt.is_positive(), 0, initial_time);
265  TimeId next_time_id = time_stepper.next_time_id(time_id, initial_dt);
266  if (next_time_id.is_at_slab_boundary()) {
267  const auto next_time = next_time_id.step_time();
268  next_time_id = TimeId(
269  initial_dt.is_positive(), 1,
270  next_time.with_slab(next_time.slab().advance_towards(initial_dt)));
271  }
272 
273  const size_t num_grid_points =
274  db::get<Tags::Mesh<Dim>>(box).number_of_grid_points();
275  const auto& inertial_coords =
276  db::get<Tags::Coordinates<Tags::ElementMap<Dim>,
277  Tags::LogicalCoordinates<Dim>>>(box);
278 
279  // Will be overwritten before use
280  DtVars dt_vars{num_grid_points};
281 
282  typename Tags::HistoryEvolvedVariables<variables_tag,
283  dt_variables_tag>::type history;
284  using solution_tag = CacheTags::AnalyticSolutionBase;
285  if (not time_stepper.is_self_starting()) {
286  // We currently just put initial points at past slab boundaries.
287  Time past_t = initial_time;
288  TimeDelta past_dt = initial_dt;
289  for (size_t i = time_stepper.number_of_past_steps(); i > 0; --i) {
290  past_dt = past_dt.with_slab(past_dt.slab().advance_towards(-past_dt));
291  past_t -= past_dt;
292  Vars hist_vars{num_grid_points};
293  DtVars dt_hist_vars{num_grid_points};
294  hist_vars.assign_subset(Parallel::get<solution_tag>(cache).variables(
295  inertial_coords, past_t.value(), typename Vars::tags_list{}));
296  dt_hist_vars.assign_subset(
297  Parallel::get<solution_tag>(cache).variables(
298  inertial_coords, past_t.value(),
299  typename DtVars::tags_list{}));
300  history.insert_initial(past_t, std::move(hist_vars),
301  std::move(dt_hist_vars));
302  }
303  }
304 
305  return db::create_from<db::RemoveTags<>, simple_tags, compute_tags>(
306  std::move(box), time_id, next_time_id, initial_dt, std::move(dt_vars),
307  std::move(history));
308  }
309  };

312  template <typename Metavariables>
313  struct DgTags {
314  using temporal_id_tag = typename Metavariables::temporal_id;
315  using flux_comm_types = FluxCommunicationTypes<Metavariables>;
316 
317  template <typename Tag>
318  using interface_tag = Tags::Interface<Tags::InternalDirections<Dim>, Tag>;
319 
320  template <typename TagsList>
321  static auto add_mortar_data(db::DataBox<TagsList>&& box) noexcept {
322  const auto& element = db::get<Tags::Element<Dim>>(box);
323 
324  typename flux_comm_types::simple_mortar_data_tag::type mortar_data{};
325  typename Tags::Mortars<Tags::Next<temporal_id_tag>, Dim>::type
326  mortar_next_temporal_ids{};
327  const auto& temporal_id = get<temporal_id_tag>(box);
328  for (const auto& direction_neighbors : element.neighbors()) {
329  const auto& direction = direction_neighbors.first;
330  const auto& neighbors = direction_neighbors.second;
331  for (const auto& neighbor : neighbors) {
332  const auto mortar_id = std::make_pair(direction, neighbor);
333  mortar_data.insert({mortar_id, {}});
334  mortar_next_temporal_ids.insert({mortar_id, temporal_id});
335  }
336  }
337 
338  return db::create_from<
339  db::RemoveTags<>,
340  db::AddSimpleTags<typename flux_comm_types::simple_mortar_data_tag,
341  Tags::Mortars<Tags::Next<temporal_id_tag>, Dim>>>(
342  std::move(box), std::move(mortar_data),
343  std::move(mortar_next_temporal_ids));
344  }
